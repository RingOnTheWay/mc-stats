  3. 添加日期组件的 JS 逻辑

  在 <script> 里，靠近底部（已有各种函数的地方），添加一段专门用于日期组件的
  JS。按你的现有脚本结构，大致放在“数据扫描相关函数”之前即可：

  // ================== 自定义日期选择组件 ==================
  let activeDatePicker = null;
  let datePickerState = {
      year: null,
      month: null, // 0-11
      inputId: null // 'scanDate'
  };

  function openDatePicker(event) {
      event.stopPropagation();
      const wrapper = document.getElementById('scanDateWrapper');
      const input = document.getElementById('scanDate');
      const display = document.getElementById('scanDateDisplay');

      // 如果已打开则关闭
      if (activeDatePicker) {
          closeDatePicker();
          return;
      }

      // 初始日期：input 已有值则用现有值，否则用今天
      let baseDate = new Date();
      if (input.value) {
          const parts = input.value.split('-');
          if (parts.length === 3) {
              const y = parseInt(parts[0], 10);
              const m = parseInt(parts[1], 10) - 1;
              const d = parseInt(parts[2], 10);
              if (!isNaN(y) && !isNaN(m) && !isNaN(d)) {
                  baseDate = new Date(y, m, d);
              }
          }
      }
      datePickerState.year = baseDate.getFullYear();
      datePickerState.month = baseDate.getMonth();
      datePickerState.inputId = 'scanDate';

      // 创建浮层
      const picker = document.createElement('div');
      picker.className = 'date-picker';
      picker.innerHTML = buildDatePickerHtml(datePickerState.year, datePickerState.month, input.value);

      // 绑定导航与选择事件
      picker.addEventListener('click', function(e) {
          e.stopPropagation();
          const action = e.target.dataset.action;
          if (action === 'prev') {
              changeMonth(-1);
              picker.innerHTML = buildDatePickerHtml(datePickerState.year, datePickerState.month, input.value);
          } else if (action === 'next') {
              changeMonth(1);
              picker.innerHTML = buildDatePickerHtml(datePickerState.year, datePickerState.month, input.value);
          } else if (e.target.classList.contains('date-picker-day') && !e.target.classList.contains('disabled')) {
              const y = parseInt(e.target.dataset.year, 10);
              const m = parseInt(e.target.dataset.month, 10);
              const d = parseInt(e.target.dataset.day, 10);
              const iso = `${y}-${String(m + 1).padStart(2, '0')}-${String(d).padStart(2, '0')}`;
              input.value = iso;
              display.textContent = iso;
              closeDatePicker();
          }
      });

      // 插入到 wrapper 下方
      const rect = wrapper.getBoundingClientRect();
      picker.style.position = 'absolute';
      picker.style.left = rect.left + 'px';
      picker.style.top = (rect.bottom + window.scrollY) + 'px';

      document.body.appendChild(picker);
      activeDatePicker = picker;

      // 点击其他地方关闭
      document.addEventListener('click', closeDatePickerOnOutside);
  }

  function closeDatePicker() {
      if (activeDatePicker && activeDatePicker.parentNode) {
          activeDatePicker.parentNode.removeChild(activeDatePicker);
      }
      activeDatePicker = null;
      document.removeEventListener('click', closeDatePickerOnOutside);
  }

  function closeDatePickerOnOutside() {
      closeDatePicker();
  }

  function changeMonth(delta) {
      let m = datePickerState.month + delta;
      let y = datePickerState.year;
      if (m < 0) { m = 11; y--; }
      if (m > 11) { m = 0; y++; }
      datePickerState.month = m;
      datePickerState.year = y;
  }

  function buildDatePickerHtml(year, month, selectedIso) {
      const today = new Date();
      const selected = selectedIso ? new Date(selectedIso) : null;

      const firstDay = new Date(year, month, 1);
      const startWeekday = firstDay.getDay(); // 0=Sun
      const daysInMonth = new Date(year, month + 1, 0).getDate();

      const monthNames = ['1月','2月','3月','4月','5月','6月','7月','8月','9月','10月','11月','12月'];
      const weekdays = ['日','一','二','三','四','五','六'];

      let html = `
          <div class="date-picker-header">
              <button type="button" data-action="prev"><span class="material-icons">chevron_left</span></button>
              <div class="date-picker-title">${year}年 ${monthNames[month]}</div>
              <button type="button" data-action="next"><span class="material-icons">chevron_right</span></button>
          </div>
          <div class="date-picker-grid">
      `;

      // 星期标题
      weekdays.forEach(w => {
          html += `<div class="date-picker-weekday">${w}</div>`;
      });

      // 空白填充
      for (let i = 0; i < startWeekday; i++) {
          html += `<div></div>`;
      }

      // 日期
      for (let d = 1; d <= daysInMonth; d++) {
          const current = new Date(year, month, d);
          const isToday = current.toDateString() === today.toDateString();
          const isSelected = selected && current.toDateString() === selected.toDateString();

          let cls = 'date-picker-day';
          if (isToday) cls += ' today';
          if (isSelected) cls += ' selected';

          html += `<div class="${cls}" data-year="${year}" data-month="${month}" data-day="${d}">${d}</div>`;
      }

      html += '</div>';
      return html;
  }
